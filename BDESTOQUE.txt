CREATE SCHEMA IF NOT EXISTS `BANCOESTOQUE` DEFAULT CHARACTER SET utf8 ;
USE `BANCOESTOQUE` ;


CREATE TABLE IF NOT EXISTS `BANCOESTOQUE`.`CADPROD` (
  `CDPROD` SMALLINT UNSIGNED NOT NULL,
  `DESC` VARCHAR(40) NULL,
  `UN` CHAR(3) NULL,
  PRIMARY KEY (`CDPROD`));




CREATE TABLE IF NOT EXISTS `BANCOESTOQUE`.`LCTMOV` (
  `ID` SMALLINT UNSIGNED NOT NULL,
  `TIPO` CHAR(1) NULL,
  `DATA` DATE NULL,
  `QTDE` SMALLINT NULL,
  `CDPROD` SMALLINT UNSIGNED NULL,
  PRIMARY KEY (`ID`),
  CONSTRAINT `FK__CADPROD_LCTMOV`
    FOREIGN KEY (`CDPROD`)
    REFERENCES `BANCOESTOQUE`.`CADPROD` (`CDPROD`));
